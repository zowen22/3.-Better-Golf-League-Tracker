-- Cumulative team points by week
WITH weekly AS (
  SELECT m.week_number, mr.team_id,
         SUM(mr.total_points) AS week_pts
  FROM match_results mr
  JOIN matchups m ON mr.matchup_id = m.matchup_id
  WHERE m.season_id = {{ season_id }} AND m.is_bye = 0
  GROUP BY m.week_number, mr.team_id
)
SELECT
  w.week_number,
  COALESCE(NULLIF(t.team_name,''),
    p1.first_name || ' & ' || p2.first_name) AS team,
  ROUND(w.week_pts::numeric, 1) AS this_week,
  ROUND(SUM(w.week_pts) OVER (
    PARTITION BY w.team_id ORDER BY w.week_number
    ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
  )::numeric, 1) AS cumulative
FROM weekly w
JOIN teams t ON w.team_id = t.team_id
LEFT JOIN players p1 ON t.player1_id = p1.player_id
LEFT JOIN players p2 ON t.player2_id = p2.player_id
ORDER BY w.week_number, cumulative DESC;

-- Playing handicaps per player per week
SELECT m.week_number,
       p.first_name || ' ' || p.last_name AS player,
       ROUND(sc.handicap_at_time_of_play) AS playing_hcp,
       COALESCE(NULLIF(t.team_name,''),
         tp1.first_name || ' & ' || tp2.first_name) AS team
FROM scorecards sc
JOIN rounds r   ON sc.round_id  = r.round_id
JOIN matchups m ON r.matchup_id = m.matchup_id
JOIN players p  ON sc.player_id = p.player_id
JOIN teams t    ON sc.team_id   = t.team_id
LEFT JOIN players tp1 ON t.player1_id = tp1.player_id
LEFT JOIN players tp2 ON t.player2_id = tp2.player_id
WHERE m.season_id = {{ season_id }} AND m.is_bye = 0
ORDER BY m.week_number, p.last_name;

Troubleshooting Guide

  1. Compare standings above to your paper records week by week. If cumulative totals differ, note which week they first diverge.
  2. Check the handicap grid. Playing HCP here should match the commissioner's handicap list. If they differ, go to Handicap Matrix and use the Override button to set the correct index for each player.
  3. Once handicaps are corrected, re-save the affected round (Admin → Schedule → week → Edit Scores → Save) to recompute net scores and points.
  4. Use the Calc Debug page from any scorecards page (Calc Debug button) to see the exact before/after for each round.
